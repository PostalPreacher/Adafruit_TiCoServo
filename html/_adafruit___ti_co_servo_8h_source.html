27 // This header is used (instead of the stock Arduino digitalPinToTimer()
28 // in pins_arduino.h) because the latter doesn't always give precedence
29 // to 16-bit timers on certain pins; e.g. on Leonardo, pin 11 refers to
30 // TIMER0A instead of TIMER1C, which we'd prefer. This means that the more
31 // esoteric Arduino-compatibles may not work off the bat until this header
32 // file is suitably amended.

34 #ifdef TIMER1_A_PIN // Defined in known_16bit_timers.h if present
35 typedef uint16_t servoPos_t; // Servo positions are 16-bit values
36 // Min/max pulse range is the same as stock Arduino Servo library:
37 #define MIN_PULSE_WIDTH 544 // Shortest pulse sent to a servo (microsec)
38 #define MAX_PULSE_WIDTH 2400 // Longest pulse sent to a servo (microsec)
39 #else // No 16-bit timers, presumably ATtiny device
40 typedef uint8_t servoPos_t; // Servo positions are 8-bit values
41 #define TIMER1_A_PIN 1

44 #endif

49 class Adafruit_TiCoServo {
50 public:
51 #ifdef __TINY_SERVO__
52  // Behavior is slightly different for ATtiny vs other devices.
53  // Due to the limited timer resolution, servo read/write only work in
54  // 'raw' timer/counter ticks; degrees or microseconds are never used.
55  Adafruit_TiCoServo(void) : pin(-1), on(false){};
56  void attach(const int8_t pin), write(const servoPos_t pos);
61  inline servoPos_t read(void) __attribute__((always_inline)) { return *ocr; }
62 #else
63  // On other boards, read/write are available both in degrees and
64  // microseconds, consistent with original Arduino Servo library.
65  Adafruit_TiCoServo(void)
66  : pin(-1), on(false), minPulse(MIN_PULSE_WIDTH),
67  maxPulse(MAX_PULSE_WIDTH){};
68  void attach(const int8_t pin, const uint16_t min = MIN_PULSE_WIDTH,
69  const uint16_t max = MAX_PULSE_WIDTH),
70  write(const uint16_t pos);
71  uint8_t read(void);
72  uint16_t readMicroseconds(void);
73 #endif
74  void detach(void);
79  inline bool attached(void) __attribute__((always_inline)) { return on; }

81 private:
